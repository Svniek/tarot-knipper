#!/usr/bin/env python3 """ Tarotkaart Knipper — PDF, Tarot_Legging_Kaarten, Tarot_Legging_Uitleg Gebruik: python3 tarot_knipper.py """ import fitz import sys import json import base64 import ssl import io import urllib.request import urllib.error from pathlib import Path # ── Uitvoerinstellingen ─────────────────────────────── UITVOER_BREEDTE = 2160 UITVOER_HOOGTE = 3840 JPEG_KWALITEIT = 100 UITVOER_FORMAAT = "png" DOEL_DPI = 300 PDF_BASIS_DPI = 72 # ───────────────────────────────────────────────────── FORMATS = { "Tarot_Legging_Kaarten": { "breedte": 3180, "hoogte": 2252, "kaarten": [ {"naam": "VERLEDEN", "rect": ( 150, 150, 950, 1520)}, {"naam": "HEDEN", "rect": (1070, 150, 1870, 1520)}, {"naam": "TOEKOMST", "rect": (1990, 150, 2790, 1520)}, ] }, "Tarot_Legging_Uitleg": { "breedte": 3180, "hoogte": 4101, "kaarten": [ {"naam": "VERLEDEN", "rect": ( 150, 150, 950, 1520)}, {"naam": "HEDEN", "rect": (1070, 150, 1870, 1520)}, {"naam": "TOEKOMST", "rect": (1990, 150, 2790, 1520)}, ] }, } CONFIG_BESTAND = Path.home() / ".tarot_knipper_config" API_KEY_BESTAND = Path.home() / ".tarot_api_key" ssl_context = ssl.create_default_context() ssl_context.check_hostname = False ssl_context.verify_mode = ssl.CERT_NONE # ── Config ──────────────────────────────────────────── def laad_config(): try: waarden = {} for regel in CONFIG_BESTAND.read_text().splitlines(): if '=' in regel: k, v = regel.split('=') waarden[k.strip()] = float(v.strip()) ml = waarden.get('ml', 18) mr = waarden.get('mr', 6) mt = waarden.get('mt', 9) mb = waarden.get('mb', 9) gx = waarden.get('gx', 5) gy = waarden.get('gy', 5) print(f" ✓ Vorige instellingen geladen: L={ml} R={mr} B={mt} O={mb} X={gx} Y={gy}") return ml, mr, mt, mb, gx, gy except: print(" (Geen opgeslagen instellingen, standaard waarden gebruikt)") return 18, 6, 9, 9, 5, 5 def sla_config_op(ml, mr, mt, mb, gx, gy): CONFIG_BESTAND.write_text(f"ml={ml}\nmr={mr}\nmt={mt}\nmb={mb}\ngx={gx}\ngy={gy}\n") print(" 💾 Instellingen opgeslagen") def laad_api_key(): try: key = API_KEY_BESTAND.read_text().strip() if key: print(" ✓ API key geladen") return key except: pass return None def sla_api_key_op(key): API_KEY_BESTAND.write_text(key) print(" 💾 API key opgeslagen") # ── PDF Preview + kalibratie ────────────────────────── def toon_preview_pdf(pagina, marges, pdf_pad): ml, mr, mt, mb, gx, gy = marges b = pagina.rect.width h = pagina.rect.height cel_b = (b - ml - mr - 2 * gx) / 3 cel_h = (h - mt - mb - 2 * gy) / 3 zoom = 2 pix = pagina.get_pixmap(matrix=fitz.Matrix(zoom, zoom), alpha=False) tmp_doc = fitz.open() tmp_pag = tmp_doc.new_page(width=pix.width, height=pix.height) tmp_pag.insert_image(tmp_pag.rect, pixmap=pix) shape = tmp_pag.new_shape() for rij in range(3): for kolom in range(3): x0 = (ml + kolom * (cel_b + gx)) * zoom y0 = (mt + rij * (cel_h + gy)) * zoom shape.draw_rect(fitz.Rect(x0, y0, x0 + cel_b * zoom, y0 + cel_h * zoom)) shape.finish(color=(1, 0, 0), width=3, fill=None) shape.commit() preview_pad = pdf_pad.parent / "preview_snijraster.jpg" tmp_doc[0].get_pixmap(matrix=fitz.Matrix(1, 1), alpha=False).save(str(preview_pad)) tmp_doc.close() print(f"\n 📸 Preview: {preview_pad}") print(" Open dit bestand om de snijlijnen te controleren.") def vraag_marges(pagina, pdf_pad, marges): b = pagina.rect.width h = pagina.rect.height print(f"\nPagina afmetingen: {b:.1f} × {h:.1f} punten") ml, mr, mt, mb, gx, gy = marges while True: cel_b = (b - ml - mr - 2 * gx) / 3 cel_h = (h - mt - mb - 2 * gy) / 3 print(f"\nHuidige marges:") print(f" Links={ml} Rechts={mr} Boven={mt} Onder={mb} Tussen-X={gx} Tussen-Y={gy}") print(f" → Celgrootte: {cel_b:.1f} × {cel_h:.1f} punten") toon_preview_pdf(pagina, (ml, mr, mt, mb, gx, gy), pdf_pad) print("\nOpties:") print(" [Enter] → Goed! Doorgaan") print(" l=20 r=10 ... → Waarden aanpassen (l r b o x y)") invoer = input("\nInvoer: ").strip().lower() if invoer == "": sla_config_op(ml, mr, mt, mb, gx, gy) break try: for deel in invoer.split(): if '=' in deel: k, v = deel.split('=') v = float(v) if k == 'l': ml = v elif k == 'r': mr = v elif k == 'b': mt = v elif k == 'o': mb = v elif k == 'x': gx = v elif k == 'y': gy = v except Exception as e: print(f" Fout: {e}") return ml, mr, mt, mb, gx, gy # ── AI Naamherkenning ───────────────────────────────── def stuur_api_request(b64, media_type, aantal, api_key): tekst = ( "Dit is een pagina met 9 tarotkaarten in een 3x3 grid. Elke kaart heeft onderaan een naam. " "Geef ALLEEN de 9 namen terug, van links naar rechts en van boven naar beneden. " "Één naam per regel, geen nummers, geen extra tekst, geen lege regels." if aantal == 9 else "Dit is een tarotlegging met 3 kaarten naast elkaar. Elke kaart heeft een naam onderaan. " "Geef ALLEEN de 3 namen terug van links naar rechts. " "Één naam per regel, geen nummers, geen extra tekst." ) payload = json.dumps({ "model": "claude-opus-4-5", "max_tokens": 200, "messages": [{ "role": "user", "content": [ {"type": "image", "source": {"type": "base64", "media_type": media_type, "data": b64}}, {"type": "text", "text": tekst} ] }] }).encode() req = urllib.request.Request( "https://api.anthropic.com/v1/messages", data=payload, headers={ "Content-Type": "application/json", "x-api-key": api_key, "anthropic-version": "2023-06-01" } ) try: with urllib.request.urlopen(req, timeout=30, context=ssl_context) as resp: data = json.loads(resp.read()) namen = [r.strip() for r in data["content"][0]["text"].strip().splitlines() if r.strip()][:aantal] while len(namen) < aantal: namen.append("") return namen except urllib.error.HTTPError as e: print(f" ❌ API fout {e.code}: {e.read().decode()[:200]}") except Exception as e: print(f" ❌ Fout: {e}") return None def detecteer_namen_pdf(pagina, api_key): print(" 🤖 Claude analyseert kaartnamen...") zoom = min(1200 / pagina.rect.width, 1.5) pix = pagina.get_pixmap(matrix=fitz.Matrix(zoom, zoom), alpha=False) b64 = base64.b64encode(pix.tobytes("jpeg")).decode() return stuur_api_request(b64, "image/jpeg", 9, api_key) def detecteer_namen_jpeg(img_pad, api_key): print(" 🤖 Claude analyseert kaartnamen...") with open(img_pad, "rb") as f: b64 = base64.b64encode(f.read()).decode() media_type = "image/jpeg" if img_pad.suffix.lower() in (".jpg", ".jpeg") else "image/png" return stuur_api_request(b64, media_type, 3, api_key) # ── PDF verwerken ───────────────────────────────────── def verwerk_pdf(pdf_pad, uitvoer_map, marges, api_key=None): doc = fitz.open(pdf_pad) ml, mr, mt, mb, gx, gy = marges uitvoer_map.mkdir(parents=True, exist_ok=True) try: from PIL import Image gebruik_pil = True except ImportError: gebruik_pil = False totaal = 0 for pagina_nr in range(len(doc)): pagina = doc[pagina_nr] b = pagina.rect.width h = pagina.rect.height cel_b = (b - ml - mr - 2 * gx) / 3 cel_h = (h - mt - mb - 2 * gy) / 3 zoom = max(DOEL_DPI / PDF_BASIS_DPI, UITVOER_BREEDTE / cel_b, UITVOER_HOOGTE / cel_h) print(f"\nPagina {pagina_nr+1}/{len(doc)} — zoom {zoom:.2f}× ({zoom * PDF_BASIS_DPI:.0f} DPI)") namen = None if api_key: namen = detecteer_namen_pdf(pagina, api_key) if namen: print(f" Namen: {', '.join(n for n in namen if n)}") for rij in range(3): for kolom in range(3): kaart_nr = rij * 3 + kolom + 1 x0 = ml + kolom * (cel_b + gx) y0 = mt + rij * (cel_h + gy) rect = fitz.Rect(x0, y0, x0 + cel_b, y0 + cel_h) pix = pagina.get_pixmap(matrix=fitz.Matrix(zoom, zoom), clip=rect, alpha=False) if namen and namen[kaart_nr - 1]: veilig = "".join(c if c not in r'/\:*?"<>|' else '_' for c in namen[kaart_nr - 1]) bestand = uitvoer_map / f"{veilig}.{UITVOER_FORMAAT}" else: bestand = uitvoer_map / f"p{pagina_nr+1}_kaart{kaart_nr:02d}.{UITVOER_FORMAAT}" if gebruik_pil and (pix.width != UITVOER_BREEDTE or pix.height != UITVOER_HOOGTE): img = Image.open(io.BytesIO(pix.tobytes("jpeg"))) img = img.resize((UITVOER_BREEDTE, UITVOER_HOOGTE), Image.LANCZOS) if UITVOER_FORMAAT == "png": img.save(str(bestand), "PNG") else: img.save(str(bestand), "JPEG", quality=JPEG_KWALITEIT, subsampling=0) else: pix.save(str(bestand)) kb = bestand.stat().st_size // 1024 print(f" ✓ Kaart {kaart_nr}/9 → {bestand.name} ({kb} KB)") totaal += 1 doc.close() print(f"\n✅ Klaar! {totaal} kaarten opgeslagen in: {uitvoer_map}") # ── JPEG legging verwerken ──────────────────────────── def verwerk_jpeg_legging(img_pad, uitvoer_map, format_key, api_key=None): try: from PIL import Image except ImportError: print("❌ Pillow niet gevonden — installeer met: pip3 install Pillow") return img = Image.open(img_pad).convert("RGB") W, H = img.size schaal = W / FORMATS[format_key]["breedte"] print(f"\nAfbeelding: {img_pad.name} — {W}×{H} px — schaal: {schaal:.3f}×") uitvoer_map.mkdir(parents=True, exist_ok=True) kaart_namen = [r["naam"] for r in FORMATS[format_key]["kaarten"]] if api_key: herkend = detecteer_namen_jpeg(img_pad, api_key) if herkend: print(f" Namen: {', '.join(herkend)}") kaart_namen = herkend for i, kaart in enumerate(FORMATS[format_key]["kaarten"]): x0 = round(kaart["rect"][0] * schaal) y0 = round(kaart["rect"][1] * schaal) x1 = round(kaart["rect"][2] * schaal) y1 = round(kaart["rect"][3] * schaal) gesneden = img.crop((x0, y0, x1, y1)) geschaald = gesneden.resize((UITVOER_BREEDTE, UITVOER_HOOGTE), Image.LANCZOS) naam = kaart_namen[i] if i < len(kaart_namen) else kaart["naam"] veilig = "".join(c if c not in r'/\:*?"<>|' else '_' for c in naam) bestand = uitvoer_map / f"{veilig}.{UITVOER_FORMAAT}" if UITVOER_FORMAAT == "png": geschaald.save(str(bestand), "PNG") else: geschaald.save(str(bestand), "JPEG", quality=JPEG_KWALITEIT, subsampling=0) kb = bestand.stat().st_size // 1024 print(f" ✓ {naam} → {bestand.name} ({kb} KB)") print(f"\n✅ Klaar! 3 kaarten opgeslagen in: {uitvoer_map}") # ── Main ────────────────────────────────────────────── def main(): huidige_map = Path.cwd() pdfs = sorted(huidige_map.glob("*.pdf")) jpegs = sorted( list(huidige_map.glob("*.jpg")) + list(huidige_map.glob("*.jpeg")) + list(huidige_map.glob("*.JPG")) + list(huidige_map.glob("*.JPEG")) ) bestanden = pdfs + jpegs if not bestanden: print(f"Geen bestanden gevonden in: {huidige_map}") sys.exit(1) if len(bestanden) == 1: bestand = bestanden[0] else: print("Bestanden gevonden:") for i, f in enumerate(bestanden): soort = "PDF" if f.suffix.lower() == ".pdf" else "JPEG" print(f" {i+1}. [{soort}] {f.name}") bestand = bestanden[int(input("Welk nummer? ").strip()) - 1] # API key api_key = laad_api_key() if not api_key: keuze = input("\n🤖 Wil je kaartnamen automatisch herkennen via AI? (j/n): ").strip().lower() if keuze == 'j': api_key = input(" Plak je Anthropic API key (sk-ant-...): ").strip() if api_key: sla_api_key_op(api_key) else: keuze = input("\n🤖 AI naamherkenning gebruiken? (j/n): ").strip().lower() if keuze == 'n': api_key = None uitvoer_map = huidige_map / f"kaarten_{bestand.stem}" if bestand.suffix.lower() == ".pdf": print(f"\nGeladen: {bestand.name} (PDF)") doc = fitz.open(bestand) marges = laad_config() marges = vraag_marges(doc[0], bestand, marges) doc.close() verwerk_pdf(bestand, uitvoer_map, marges, api_key) else: # Detecteer format op naam, anders vragen format_key = next((k for k in FORMATS if k in bestand.name), None) if not format_key: print(f"\nWelk format is '{bestand.name}'?") for i, k in enumerate(FORMATS.keys()): print(f" {i+1}. {k}") keuze = int(input("Keuze: ").strip()) - 1 format_key = list(FORMATS.keys())[keuze] verwerk_jpeg_legging(bestand, uitvoer_map, format_key, api_key) input("\nDruk op Enter om af te sluiten...") if __name__ == "__main__": main()
